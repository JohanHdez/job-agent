## File Naming Convention

- TypeScript/Node.js: `[module].test.ts` — **never** `.spec.ts`
- React components: `[Component].test.tsx` — **never** `.spec.tsx`
- Co-located with source: `src/utils/logger.test.ts` next to `src/utils/logger.ts`

## Test Structure to Implement

### `packages/core/` — Type Guards & Validators
```
packages/core/src/types/
├── config.types.test.ts         # validateConfig() with valid/invalid inputs
├── cv.types.test.ts             # ProfessionalProfileType shape validation
└── job.types.test.ts            # ApplicationStatusType exhaustiveness
```

**Pattern:**
```typescript
import { describe, it, expect } from 'vitest';
import { validateConfig } from './config.types';

describe('validateConfig', () => {
  it('throws on missing required fields', () => {
    expect(() => validateConfig({})).toThrow();
  });

  it('returns typed config for valid input', () => {
    const result = validateConfig({ search: { keywords: ['TypeScript'] } });
    expect(result.search.keywords).toContain('TypeScript');
  });
});
```

**NestJS test pattern:**
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import * as request from 'supertest';

describe('AuthController (e2e)', () => {
  let app: INestApplication;

  beforeEach(async () => {
    const module: TestingModule = await Test.createTestingModule({
      imports: [AppModule],
    }).compile();
    app = module.createNestApplication();
    await app.init();
  });

  it('/auth/login (POST)', () => {
    return request(app.getHttpServer())
      .post('/auth/login')
      .send({ email: 'test@test.com', password: 'password' })
      .expect(200);
  });
});
```

## Vitest Configuration (To Be Added)

Add to each `packages/*/package.json`:
```json
{
  "scripts": {
    "test": "vitest run",
    "test:watch": "vitest",
    "test:coverage": "vitest run --coverage"
  },
  "devDependencies": {
    "vitest": "^1.0.0",
    "@vitest/coverage-v8": "^1.0.0"
  }
}
```

Add `vitest.config.ts` to each package:
```typescript
import { defineConfig } from 'vitest/config';

export default defineConfig({
  test: {
    coverage: {
      reporter: ['text', 'lcov'],
      thresholds: { lines: 80, functions: 80 },
    },
  },
});
```

    "rootDir": "src",
    "testRegex": ".*\\.test\\.ts$",
    "transform": { "^.+\\.(t|j)s$": "ts-jest" },
    "coverageDirectory": "../coverage",
    "testEnvironment": "node",
    "coverageThreshold": {
