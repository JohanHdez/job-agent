**Worker processor logic:**

```typescript
async function processSession(job: Job<{ sessionId: string; userId: string }>): Promise<void> {
  const { sessionId, userId } = job.data;
  const channel = `session:${sessionId}:events`;

  // 1. Update session status to 'running' and set startedAt
  await SessionModel.updateOne({ _id: sessionId }, { $set: { status: 'running', startedAt: new Date() } });

  // 2. Generate mock events
  const events = generateMockSessionEvents(sessionId, userId);
  const startTime = Date.now();

  // 3. For each event, with 500-1500ms delay between events:
  for (const event of events) {
    // Check if session was cancelled
    const session = await SessionModel.findById(sessionId, { status: 1 });
    if (session?.status === 'cancelled') {
      break;
    }

    // Persist event to MongoDB (ring buffer)
    const updated = await SessionModel.findByIdAndUpdate(
      sessionId,
      { $inc: { nextEventId: 1 } },
      { new: true, projection: { nextEventId: 1 } }
    );
    const eventId = updated!.nextEventId;

    const storedEvent = { id: eventId, type: event.type, data: event as Record<string, unknown>, timestamp: event.timestamp };

    await SessionModel.updateOne(
      { _id: sessionId },
      { $push: { events: { $each: [storedEvent], $slice: -100 } } }
    );

    // Publish to Redis Pub/Sub for live SSE consumers
    await publisher.publish(channel, JSON.stringify(storedEvent));

    // Delay between events (simulate real search work)
    await new Promise(resolve => setTimeout(resolve, 500 + Math.random() * 1000));
  }

  // 4. Update session status to 'completed'
  await SessionModel.updateOne(
    { _id: sessionId },
    { $set: { status: 'completed', completedAt: new Date() } }
  );
}
```

**Update `apps/api/src/modules/sessions/sessions.module.ts`:**

Add `OnModuleInit` and `OnModuleDestroy` lifecycle hooks to spawn and manage the worker process.

Use `child_process.fork` to spawn the compiled worker JS file from the NestJS module:

```typescript
import { Module, OnModuleInit, OnModuleDestroy } from '@nestjs/common';
import { fork, ChildProcess } from 'child_process';
import { join } from 'path';
import { BullModule } from '@nestjs/bullmq';
import { MongooseModule } from '@nestjs/mongoose';
import { Session, SessionSchema } from './schemas/session.schema.js';
import { SessionsService } from './sessions.service.js';
import { SessionsController } from './sessions.controller.js';

/**
 * Sessions module — manages session lifecycle, BullMQ queue, and worker process.
 *
 * The worker is spawned as a separate Node.js process via child_process.fork
 * for crash isolation (locked decision: worker crash must not affect API).
 */
@Module({
  imports: [
    BullModule.registerQueue({ name: 'search-session' }),
    MongooseModule.forFeature([{ name: Session.name, schema: SessionSchema }]),
  ],
  controllers: [SessionsController],
  providers: [SessionsService],
  exports: [SessionsService],
})
export class SessionsModule implements OnModuleInit, OnModuleDestroy {
  private worker: ChildProcess | null = null;

  /** Spawn the BullMQ worker as a separate process on module init */
  onModuleInit(): void {
    const workerPath = join(__dirname, '../../workers/search-session.worker.js');
    this.worker = fork(workerPath, [], {
      env: { ...process.env },
      stdio: ['ignore', 'pipe', 'pipe', 'ipc'],
    });

    this.worker.stdout?.on('data', (data: Buffer) => {
      process.stdout.write(`[worker:stdout] ${data.toString()}`);
    });
    this.worker.stderr?.on('data', (data: Buffer) => {
      process.stderr.write(`[worker:stderr] ${data.toString()}`);
    });

    this.worker.on('exit', (code: number | null) => {
      console.warn(`[SessionsModule] Worker exited with code ${code}`);
      this.worker = null;
      // In production, add respawn logic here
    });
  }

  /** Kill the worker process on module destroy (graceful shutdown) */
  onModuleDestroy(): void {
    if (this.worker) {
      this.worker.kill('SIGTERM');
      this.worker = null;
    }
  }
}
```

The `workerPath` resolves relative to the compiled `dist/` output. After `nest build`, the worker file will be at `dist/apps/api/src/workers/search-session.worker.js` relative to project root, which maps to `../../workers/search-session.worker.js` relative to the compiled `sessions.module.js` in `dist/.../modules/sessions/`.
