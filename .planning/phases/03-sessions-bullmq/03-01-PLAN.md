**3. Create `apps/api/src/modules/sessions/schemas/session.schema.ts`:**

Follow the User schema pattern exactly (Prop, Schema, SchemaFactory from @nestjs/mongoose):

```typescript
import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { HydratedDocument } from 'mongoose';
import type { SessionStatus, StoredSessionEvent } from '@job-agent/core';

export type SessionDocument = HydratedDocument<Session>;

@Schema({ timestamps: true, collection: 'sessions' })
export class Session {
  @Prop({ required: true, index: true })
  userId!: string;

  @Prop({ required: true, enum: ['queued', 'running', 'completed', 'cancelled', 'failed'], default: 'queued' })
  status!: SessionStatus;

  @Prop({ type: Object, required: true })
  config!: Record<string, unknown>;  // SearchConfigSnapshot at start time

  @Prop({ type: [Object], default: [] })
  events!: StoredSessionEvent[];     // ring buffer — max 100 via $push + $slice

  @Prop({ type: Number, default: 0 })
  nextEventId!: number;              // monotonic counter

  @Prop()
  startedAt?: Date;

  @Prop()
  completedAt?: Date;
}

export const SessionSchema = SchemaFactory.createForClass(Session);

// Compound indexes for per-user queries
SessionSchema.index({ userId: 1, status: 1 });
SessionSchema.index({ userId: 1, createdAt: -1 });
```

**1. Create `apps/api/src/common/redis/redis-subscriber.provider.ts`:**

```typescript
import Redis from 'ioredis';

/** DI injection token for the dedicated Redis Pub/Sub subscriber client */
export const REDIS_SUBSCRIBER = 'REDIS_SUBSCRIBER';

/**
 * Dedicated ioredis instance for Pub/Sub subscriptions.
 * MUST be a separate connection from REDIS_CLIENT — once subscribe() is called,
 * the connection enters subscriber mode and cannot run other Redis commands.
 */
export const RedisSubscriberProvider = {
  provide: REDIS_SUBSCRIBER,
  useFactory: (): Redis =>
    new Redis(process.env['REDIS_URL'] ?? 'redis://localhost:6379'),
};
```
