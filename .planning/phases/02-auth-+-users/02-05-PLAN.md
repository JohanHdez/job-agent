<!-- Backend endpoints this plan depends on (created by Plan 03) -->

GET /users/profile:
  Response: { profile: ProfessionalProfile | null, isComplete: boolean, missingFields: string[] }

PATCH /users/profile:
  Request: Partial<ProfessionalProfile> (UpdateProfileDto)
  Response: Updated user document

POST /users/profile/cv:
  Request: multipart form-data with 'cv' field (PDF)
  Response: { profile: ProfessionalProfile, isComplete: boolean, missingFields: string[] }

GET /users/presets:
  Response: SearchPresetType[]

POST /users/presets:
  Request: CreatePresetDto (name, keywords, location, modality, platforms, seniority, languages, datePosted, minScoreToApply, maxApplicationsPerSession, excludedCompanies)
  Response: SearchPresetType (with generated id)
  Error 400: "Maximum 5 presets reached. Delete one first."

PATCH /users/presets/active:
  Request: { presetId: string }
  Response: Updated user

DELETE /users/presets/:id:
  Response: Updated user
