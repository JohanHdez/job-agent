/** Status of a vacancy in the user's history */

export interface SearchConfigSnapshotType {
  keywords: string[];
  location: string;
  modality: ('Remote' | 'Hybrid' | 'On-site')[];
  platforms: PlatformId[];
  seniority: string[];
  languages: string[];
  datePosted: 'past_24h' | 'past_week' | 'past_month';
  minScoreToApply: number;
  maxApplicationsPerSession: number;
  excludedCompanies: string[];
}

/** A job vacancy discovered during a search session, persisted in the vacancies collection */
export interface VacancyType {
  /** JSearch job_id or platform-specific unique ID */
  jobId: string;
  title: string;
  company: string;
  description: string;
  url: string;
  location: string;
  platform: PlatformId;
  postedAt: string;
  /** 0-100 compatibility score from the scoring engine */
  compatibilityScore: number;
  /** Reason string from the LLM scorer (max 15 words) */
  scoreReason: string;
  /** Current status in the user's history */
  status: VacancyStatus;
  /** MongoDB ObjectId string of the owning user (NF-08 row-level security) */
  userId: string;
  /** MongoDB ObjectId string of the session that discovered this vacancy */
  sessionId: string;
  /** ISO 8601 timestamp when the vacancy was first discovered */
  discoveredAt: string;
  /** Why this vacancy was filtered out locally, if applicable */
  filterReason?: 'missing_fields' | 'excluded_company' | 'duplicate' | 'score_below_threshold';
}

/** Parameters passed to a job search adapter */
export interface SearchParams {
  keywords: string[];
  location: string;
  modality: ('Remote' | 'Hybrid' | 'On-site')[];
  datePosted: 'past_24h' | 'past_week' | 'past_month';
  /** Minimum number of results to collect before stopping pagination */
  minResults: number;
  /** Maximum pages to fetch (safety limit) */
  maxPages: number;
}

/** Raw job result returned by a search adapter before scoring */
export interface RawJobResult {
  /** Platform-specific unique job identifier */
  jobId: string;
  title: string;
  company: string;
  description: string;
  url: string;
  location: string;
  platform: PlatformId;
  /** ISO 8601 date string when the job was posted */
  postedAt: string;
}

/** Interface for swappable job search providers (JSearch, Adzuna, etc.) */
export interface JobSearchAdapter {
  /** Human-readable name of the provider (e.g. "JSearch (RapidAPI)") */
  readonly name: string;
  /** Platform ID this adapter searches */
  readonly platform: PlatformId;
  /** Search for jobs matching the given parameters. Returns raw results before scoring. */
  search(params: SearchParams): Promise<RawJobResult[]>;
}

**vacancy.schema.ts:**
```typescript
import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { HydratedDocument } from 'mongoose';
import type { VacancyStatus } from '@job-agent/core';

export type VacancyDocument = HydratedDocument<Vacancy>;

@Schema({ timestamps: true, collection: 'vacancies' })
export class Vacancy {
  @Prop({ required: true, index: true })
  jobId!: string;

  @Prop({ required: true })
  title!: string;

  @Prop({ required: true })
  company!: string;

  @Prop({ required: true })
  description!: string;

  @Prop({ required: true })
  url!: string;

  @Prop({ required: true })
  location!: string;

  @Prop({ required: true })
  platform!: string;

  @Prop()
  postedAt?: string;

  @Prop({ required: true, min: 0, max: 100 })
  compatibilityScore!: number;

  @Prop({ default: '' })
  scoreReason!: string;

  @Prop({ required: true, enum: ['new', 'applied', 'dismissed', 'failed'], default: 'new' })
  status!: VacancyStatus;

  @Prop({ required: true, index: true })
  userId!: string;

  @Prop({ required: true, index: true })
  sessionId!: string;

  @Prop({ required: true })
  discoveredAt!: string;

  @Prop()
  filterReason?: string;
}

export const VacancySchema = SchemaFactory.createForClass(Vacancy);

// Deduplication indexes (per-user scope)
/** Unique URL per user — prevents same job URL appearing twice for the same user */
VacancySchema.index({ userId: 1, url: 1 }, { unique: true });
/** Company+title compound per user — catches duplicate postings across platforms */
VacancySchema.index(
  { userId: 1, company: 1, title: 1 },
  { unique: true, collation: { locale: 'en', strength: 2 } }
);
/** Session lookup — list all vacancies for a session */
VacancySchema.index({ sessionId: 1, compatibilityScore: -1 });
/** User history — list all vacancies for a user sorted by discovery time */
VacancySchema.index({ userId: 1, discoveredAt: -1 });
```
